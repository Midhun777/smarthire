1. Database Table Definitions

Table: tbl_users

Sl noFieldTypeDescription
1user_idINT AUTO_INCREMENTPrimary Key
2full_nameVARCHAR(100)User's full name
3emailVARCHAR(150)Unique email address
4roleVARCHAR(20)Enum: admin, job_seeker, job_provider
5password_hashVARCHAR(255)Hashed password

Table: tbl_seeker_profiles

Sl noFieldTypeDescription
1profile_idINT AUTO_INCREMENTPrimary Key
2user_idINTFK → tbl_users.user_id
3resume_textTEXTParsed resume content
4skillsTEXTComma‑separated skill list

Table: tbl_job_listings

Sl noFieldTypeDescription
1job_idINT AUTO_INCREMENTPrimary Key
2user_idINTFK → tbl_users.user_id (poster)
3titleVARCHAR(150)Job title
4descriptionTEXTJob description
5category_idINTFK → tbl_categories.category_id

Table: tbl_applications

Sl noFieldTypeDescription
1app_idINT AUTO_INCREMENTPrimary Key
2user_idINTFK → tbl_users.user_id (applicant)
3job_idINTFK → tbl_job_listings.job_id
4statusVARCHAR(20)Enum: Pending, Reviewed, Accepted, Rejected
5applied_atDATETIMETimestamp of application

Table: tbl_categories

Sl noFieldTypeDescription
1category_idINT AUTO_INCREMENTPrimary Key
2nameVARCHAR(100)Category name (e.g., IT, Finance)
